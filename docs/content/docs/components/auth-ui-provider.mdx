title: <AuthUIProvider>
---

The `<AuthUIProvider>` wraps your application with authentication context, providing essential hooks, settings, and methods required by authentication-related components and hooks throughout your app.

Basic usage:

```tsx title="providers.tsx"
"use client"

import { AuthUIProvider } from '@daveyplate/better-auth-ui'
import { authClient } from '@/lib/auth-client'
import { useRouter } from 'next/navigation'
import Link from 'next/link'

export function Providers({ children }: { children: React.ReactNode }) {
  const router = useRouter()

  return (
    <AuthUIProvider
      authClient={authClient}
      navigate={router.push}
      replace={router.replace}
      onSessionChange={() => router.refresh()}
      LinkComponent={Link}
    >
      {children}
    </AuthUIProvider>
  )
}
```

## Example

A minimal Next.js layout file using the `AuthUIProvider`:

```tsx
"use client"

import { AuthUIProvider } from "@daveyplate/better-auth-ui"
import { authClient } from "@/lib/auth-client"
import { useRouter } from "next/navigation"
import Link from "next/link"

export default function RootLayout({ children }: { children: React.ReactNode }) {
  const router = useRouter()

  return (
    <html lang="en">
      <body>
        <AuthUIProvider
          authClient={authClient}
          navigate={router.push}
          replace={router.replace}
          onSessionChange={() => router.refresh()}
          providers={["google", "github"]} // multiple providers supported
          credentials={true}
          multiSession={true} // enable multiple sessions per device
          username={true} // enable username sign-in
          magicLink={true}
          passkey={true}
          avatar={true}
          uploadAvatar={(file) => {
            const formData = new FormData()
            formData.append("avatar", file)
            return fetch("/api/uploadAvatar", { method: "POST", body: formData })
              .then((res) => res.json())
              .then((data) => data.avatarUrl)
          }}
          settingsUrl="/user/settings"
          LinkComponent={Link}
        >
          {children}
        </AuthUIProvider>
      </body>
    </html>
  )
}
```
